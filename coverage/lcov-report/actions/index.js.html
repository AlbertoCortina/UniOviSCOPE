export const DELETE_ERROR = 'DELETE_ERROR'

export const AUTHENTICATE = 'AUTHENTICATE'

export const LANDING_SCREEN_HOME_SCREEN = 'LANDING_SCREEN_HOME_SCREEN'

        username,
        firstname,
        lastname,

    }
